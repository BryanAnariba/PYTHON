CREAR ENTORNO VIRTUAL Y ACTIVAR
    python -m venv env
    .\env\Scripts\activate

INSTALAR DJANGO => dentro del entorno virtual claro
    pip install Django==4.1.3

CREAR PROYECTO
    django-admin startproject nombre_proyecto

VERIFICAR PAQUETES INSTALADO
    pip list

EJECUTAR SERVER => dentro del entorno virtual claro
    python manage.py runserver
